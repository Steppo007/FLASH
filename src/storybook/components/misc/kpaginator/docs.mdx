## Example Usage

The paginator above was produced with the following code. You should 
imagine there is a table with 120 rows, and these rows are split into
pages. The dropdown controls how many rows appear in each page.

```jsx
import { KPaginator } from '@flash/ui-components/misc/kpaginator';
import { PaginatorPageChangeEvent } from 'primereact/paginator';
import { useState } from 'react';

export function Example() {
  const [first, setFirst] = useState(0);
  const [rows, setRows] = useState(10);

  const onPageChange = (event: PaginatorPageChangeEvent) => {
    setFirst(event.first);
    setRows(event.rows);
  };

  return (
    <KPaginator
      first={first}
      rows={rows}
      totalRecords={120}
      rowsPerPageOptions={[10, 20, 30]}
      onPageChange={onPageChange}
    />
  );
}
```
