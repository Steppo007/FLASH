### Uses: [PrimeReact Tab Menu](https://primereact.org/tabmenu)

The KTabMenu is a navigation component that displays menu items as tab headers.

# Usage

<Example/>

The `KTabMenu`, like the `KSideMenu`, is derived from the PrimeReact `TabMenu`.
The two components are therefore extremely similar. The only differences are:
- The styling is different (for example, one is horizontal, the other vertical)
- The `KTabMenu` has support for adding badges
- The `KTabMenu` has support for `disabled` tabs
- The `KTabMenu` also has a `size` attribute

export const model: MenuItem[] = [
  {
    label: 'Account',
    icon: <AccountFigure />,
    id: 'account',
    data: {
      badge: <KBadge label="1" severity="danger" size="small" />,
    },
  },
  {
    label: 'Theme',
    icon: <ThemePalette />,
    id: 'theme',
  },
  {
    label: 'Language',
    icon: <LanguageGlobe />,
    id: 'language',
    disabled: true,
  },
];

export default function Example() {
  const [activeId, setActiveId] = useState('');
  return (
    <ExampleWrapper className="flex justify-center">
      <KTabMenu
        model={model}
        activeIndex={model.findIndex((x) => x.id === activeId)}
        onTabChange={(e) => {
          setActiveId(e.value.id ?? '');
        }}
        size="default"
        className="w-full"
      />
    </ExampleWrapper>
  );
}
