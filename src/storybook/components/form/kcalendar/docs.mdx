### Uses: [PrimeReact Calendar](https://primereact.org/calendar)

The KCalendar allows users to input date selections and ranges.

<Canvas of={metaData.Story} />


<Controls />

## Examples

**Important**: Note that `onChange` and `value` behave differently for different selection modes.
If `selectionMode === 'single'`, then the value of the KCalendar will be a single date, otherwise the value will be an array of dates.
For more information, look at the following examples.

### Example: Single selection

<ExampleSingleSelection />

[comment]: # (START WRITE ./exampleSingle.tsx)

```jsx
import { KCalendar } from '@flash/ui-components/form/kcalendar';
import { ExampleWrapper } from '@root/src/storybook/utils';
import { FormEvent } from 'primereact/ts-helpers';
import { useState } from 'react';

export function ExampleSingleSelection() {
  const [value, setValue] = useState<Date>();

  return (
    <ExampleWrapper className="flex justify-center">
      <KCalendar
        value={value}
        selectionMode="single"
        floatLabel="Select a date"
        helperText="Helper text"
        onChange={(e: FormEvent<Date>) => {
          setValue(e.value ?? undefined);
        }}
      />
    </ExampleWrapper>
  );
}
```
