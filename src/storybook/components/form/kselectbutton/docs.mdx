### Uses: [PrimeReact Select Button](https://primereact.org/selectbutton)

The KSelectButton is used to choose single or multiple items from a list using buttons.

## Example Usage

Use the KSelectButton exactly as one uses the PrimeReact [Select Button](https://primereact.org/selectbutton). 

### Using Text Labels
<TextExample/>

[comment]: # (START WRITE ./textExample.tsx)
[comment]: # (DO NOT EDIT: the following has been auto-generated with the command "npm run sync:docs")
```jsx
import { KSelectButton } from '@flash/ui-components/form/kselectbutton';
import { ExampleWrapper } from '@root/src/storybook/utils';
import { useState } from 'react';

export function TextExample() {
  const textOptions = [
    { label: 'Small', value: 'sm' },
    { label: 'Medium', value: 'md' },
    { label: 'Large', value: 'lg' },
  ];

  const [value, setValue] = useState(textOptions[0].value);

  return (
    <ExampleWrapper className="flex justify-center">
      <KSelectButton
        allowEmpty={false}
        value={value}
        onChange={(e) => setValue(e.value)}
        options={textOptions}
      />
    </ExampleWrapper>
  );
}
```
